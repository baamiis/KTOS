229 static void __attribute__((unused)) ktos_SwitchTask(void)
230 {
231  static WORD Delay;
232 
233  while (TRUE)
234  {
235  ktos_hal_DisableInterrupts();
236 
237  if (MultiTask) {
238  TaskCurrent = TaskCurrent->TaskNext;
239  }
240 
241  if (TaskCurrent->Sleeping) {
242  if (TaskCurrent->TimerFlag) {
243  /* Sleep timer expired — mark the task ready. */
244  TaskCurrent->Timer = 0;
245  TaskCurrent->TimerFlag = FALSE;
246  TaskCurrent->Sleeping = FALSE;
247  }
248  }
249 
250  if (!TaskCurrent->Sleeping &&
251  (TaskCurrent->MsgCount != 0 || TaskCurrent->TimerFlag))
252  {
253  if (TaskCurrent->MsgCount != 0) {
254  /* Advance the read pointer (consume the oldest message). */
255  if (++TaskCurrent->MsgQueueOut >= TaskCurrent->MsgQueueEnd) {
256  TaskCurrent->MsgQueueOut = TaskCurrent->MsgQueue;
257  }
258  --TaskCurrent->MsgCount;
259  }
260 
261  /* Switch into the task; returns when the task yields back. */
262  ktos_hal_ContextSwitch((void **)&OS_SP, TaskCurrent->StackPtr);
263  /* Interrupts assumed disabled on return. */
264 
265  Delay = g_LastTaskReturnValue;
266 
267  if (Delay == 0) {
268  /* Yield — re-schedule immediately. */
269  TaskCurrent->TimerFlag = TRUE;
270  TaskCurrent->Timer = 0;
271  } else if (Delay == MSG_WAIT) {
272  /* Sleep indefinitely until ktos_WakeUp(). */
273  TaskCurrent->Timer = 0;
274  TaskCurrent->TimerFlag = FALSE;
275  } else {
276  /* Sleep for Delay milliseconds. */
277  TaskCurrent->Timer = Delay;
278  TaskCurrent->TimerFlag = FALSE;
279  }
280  }
281 
282  ktos_hal_EnableInterrupts();
283  }
284 }
